async function start() { startBtn.disabled = true; logDiv.innerHTML = ''; printLog('Запрос прав доступа...'); try { // ЗАМЕНИ 0 на свой ID приложения (например: 51234567) const APP_ID = 0; if (APP_ID === 0) { printLog('ОШИБКА: Замени APP_ID в коде на ID твоего приложения!', 'error'); startBtn.disabled = false; return; } const auth = await vkBridge.send("VKWebAppGetAuthToken", { "app_id": APP_ID, "scope": "friends" }); const token = auth.access_token; printLog('Токен получен. Загружаю список друзей...', 'success'); // Дальше код без изменений... const friendsData = await vkBridge.send("VKWebAppCallAPIMethod", { "method": "friends.get", "params": { "fields": "deactivated", "v": "5.131", "access_token": token } }); const allFriends = friendsData.response.items; const dogs = allFriends.filter(f => f.deactivated); printLog(`Всего друзей: ${allFriends.length}. «Собачек»: ${dogs.length}`); for (let dog of dogs) { await vkBridge.send("VKWebAppCallAPIMethod", { "method": "friends.delete", "params": { "user_id": dog.id, "v": "5.131", "access_token": token } }); printLog(`Удален ID: ${dog.id}`, 'success'); await delay(400); } printLog('Готово!', 'success'); } catch (error) { // Выводим подробности ошибки, чтобы не было undefined const errorDetail = JSON.stringify(error); printLog('Ошибка ВК: ' + errorDetail, 'error'); console.error(error); } finally { startBtn.disabled = false; } }
